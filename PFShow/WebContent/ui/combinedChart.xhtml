package org.primefaces.examples.view;

import java.io.Serializable;

import org.primefaces.model.chart.CartesianChartModel;
import org.primefaces.model.chart.ChartSeries;

public class ChartBean implements Serializable {

    private CartesianChartModel combinedModel;
    private CartesianChartModel combinedModel2;

	public ChartBean() {
        createCombinedModel();
	}

    public CartesianChartModel getCombinedModel() {
        return combinedModel;
    }

    public CartesianChartModel getCombinedModel2() {
        return combinedModel2;
    }

     private void createCombinedModel() {
        combinedModel = new CartesianChartModel();

        BarChartSeries boys = new BarChartSeries();
        boys.setLabel("Boys");

        boys.set("2004", 120);
        boys.set("2005", 100);
        boys.set("2006", 44);
        boys.set("2007", 150);
        boys.set("2008", 25);

        LineChartSeries girls = new LineChartSeries();
        girls.setLabel("Girls");

        girls.set("2004", 52);
        girls.set("2005", 60);
        girls.set("2006", 110);
        girls.set("2007", 135);
        girls.set("2008", 120);

        combinedModel.addSeries(boys);
        combinedModel.addSeries(girls);
    }
    
    private void createCombinedModel2() {
        combinedModel2 = new CartesianChartModel();

        BarChartSeries boys = new BarChartSeries();
        boys.setLabel("Boys");

        boys.set("2004", 120);
        boys.set("2005", 100);
        boys.set("2006", 44);
        boys.set("2007", 150);
        boys.set("2008", 25);

        LineChartSeries girls = new LineChartSeries();
        girls.setLabel("Girls");

        girls.set("2004", 52);
        girls.set("2005", 60);
        girls.set("2006", 110);
        girls.set("2007", 135);
        girls.set("2008", 120);
        girls.setFill(true);
        combinedModel2.addSeries(girls);
        combinedModel2.addSeries(boys);
        
    } 
}
